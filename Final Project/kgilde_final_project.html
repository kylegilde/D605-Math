inspect_metadata <- function(df){
  ###Takes a data frame &
  ###Checks for NAs, incorrect class types, and inspects the unique values
  df_len <- nrow(df)
  NA_ct = as.vector(rapply(df, function(x) sum(is.na(x))))
  var_names <- names(df)
  
  df_metadata <- data.frame(
    position = 1:length(var_names),
    var_names = var_names,
    NA_ct = NA_ct,
    NA_percent = round(NA_ct / df_len, 4),
    class_type = rapply(df, class),
    uniq_value_ct = rapply(df,function(x)length(unique(x))),
    uniq_values = rapply(df, function(x) paste(sort(unique(x)), collapse = ','))
  )
  metrics <- data.frame(describe(df))
  factor_indices <- which(df_metadata$class_type == "factor")
  metrics[factor_indices, ] <- ''
  metrics <- metrics[, 2:ncol(metrics)]
  df_summary <- cbind(df_metadata, metrics)
  return(df_summary)
}

model_summary <- function(model, model_sum, metrics_glance) {
    ### Takes as inputs some of the model outputs & creates data frame of summary
    ### statitics
    results <- data.frame(n_vars = length(names(model$model)), RootMSLE = RMSLE(model$fitted.values, 
        model$model$SalePrice), adj.r.sq = model_sum$adj.r.squared, f.statistic.and.dfs = paste(round(model_sum$fstatistic, 
        4), collapse = ", "), p.value = metrics_glance$p.value)
    return(results)
}
